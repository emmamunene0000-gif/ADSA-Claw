Build "Absolute Dollar Terminal" — a live trading intelligence terminal. This is a single fullstack Replit app: React frontend + Express backend in one project, deployed always-on.

SECTION 1 — WHAT THIS IS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

A professional trading terminal for one operator. It mirrors the ADSA v8 Pine Script agent running on TradingView — every section, every label, every input, every message format. It receives live webhook signals from TradingView, displays them in real time, and runs AI research on a watchlist of 24 assets.

The operator already has:
- TradingView Premium running ADSA v8 Pine Script on XAUUSD 1-min
- TradeSgnl Advanced executing trades on MT5 via webhooks
- Deriv broker (MT5 CFDs + Vanilla Options)
- Bybit (perpetual futures)

This terminal is the operator's mission control screen — it does not replace TradingView, it extends it.

Frontend: React 18 + TypeScript + Vite
Styling: Tailwind CSS — dark terminal, NO light mode
Charts: lightweight-charts (npm package)
Backend: Express.js (same project, runs on port 3001)
Database: Replit PostgreSQL via pg package (raw SQL only, no ORM)
AI: Anthropic Claude API (claude-sonnet-4-20250514) for watchlist research
Real-time: Server-Sent Events (SSE)
No auth, no Stripe, no multi-user complexity

SECTION 5 — BACKEND (server/index.js)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Five endpoints:

POST /webhook/signal
  Header required: x-adsa-secret (match env ADSA_SECRET)
  Body: raw text string in TradeSgnl format:
    "LICENSE_ID,XAUUSD,buy,vol_dollar=15,sl_price=4190.50,tp1_price=4205.00,pct1=0.33,tp2_price=4212.00,pct2=0.50,tp3_price=4219.00,exent=1"
  Parse into fields: symbol, action, vol_dollar, sl_price, tp1_price, pct1, tp2_price, pct2, tp3_price
  Save to signals table
  SSE broadcast: { type: "signal_new", data: parsed_signal }
  Return 200

POST /webhook/message
  Body: raw Glass Box text (the full {{alert_message}} string from TradingView)
  Detect event_type from the EVENT line: extract the string after "🔔 EVENT: "
  Save to messages table with content and event_type
  SSE broadcast: { type: "glass_box", data: { content, event_type, timestamp } }
  Return 200

POST /webhook/outcome
  Body: { trade_id, outcome_type, pts_actual }
  outcome_type: "TP1" | "TP2" | "TP3" | "SL" | "HOLDER_EXIT"
  Update signal status in db
  Update daily performance aggregate
  SSE broadcast: { type: "signal_update", data: { trade_id, outcome_type, pts_actual } }
  Return 200

GET /terminal/stream
  SSE endpoint, Content-Type: text/event-stream
  Keep alive with comment ping every 15 seconds
  All connected clients receive all broadcast events

GET /api/state
  Returns JSON:
  {
    active_signal: (latest ACTIVE signal or null),
    today_performance: { total_sigs, blocked, tp1_hits, tp2_hits, tp3_hits, sl_hits, net_pts, win_rate, profit_factor },
    recent_signals: (last 20 signals, newest first),
    recent_messages: (last 50 messages, newest first),
    dashboard: (current dashboard state object — see Section 6)
  }

POST /api/research
  Body: { symbol, question }
  Calls Claude API with system prompt (Section 9)
  Returns: { symbol, analysis, risk_verdict, key_level, session_context }

Parser function (implement in server):
parseTradeSgnlString(str):
  parts = str.split(',')
  symbol = parts[1]
  action = parts[2]  // buy/sell/modifybuy/modifysell/closebuy/closesell
  kvs = parts.slice(3) parsed as key=value
  return { symbol, action, vol_dollar, sl_price, tp1_price, pct1, tp2_price, pct2, tp3_price, exent }

formatPrice(v: number): string
  // Exact match of Pine _p() function
  const t = Math.abs(v);
  if (t >= 10000) return v.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  if (t >= 100)   return v.toFixed(2);
  if (t >= 1)     return v.toFixed(4);
  return v.toFixed(5);

formatPips(entry: number, price: number, symbol: string): string
  // For XAUUSD and indices: pts. For forex: pips (0.0001 or 0.01 for JPY)
  const dist = Math.abs(price - entry);
  if (symbol.includes('JPY')) return (dist / 0.01).toFixed(1) + ' pips';
  if (['GBPUSD','EURUSD','NZDUSD','EURAUD'].includes(symbol)) return (dist / 0.0001).toFixed(1) + ' pips';
  return dist.toFixed(2) + ' pts';

pfTag(pf: number): string
  if pf === 0: return "NO DATA"
  if pf >= 2.5: return "ELITE ✅"
  if pf >= 2.0: return "STRONG ✅"
  if pf >= 1.5: return "GOOD"
  if pf >= 1.0: return "MARGINAL ⚠️"
  return "NEGATIVE ❌"

scoreColor(score: number): string
  if score >= 6:  return "#FFD700"
  if score >= 3:  return "#00c9a7"
  if score >= 0:  return "#f4a523"
  if score >= -3: return "#6b6b8a"
  return "#ff4757"

layerIcon(score: number): string
  if score >= 2.5:  return "🟢"
  if score <= -2.5: return "🔴"
  if score > 0:     return "🟡"
  if score < 0:     return "🟠"
  return "⚪"
